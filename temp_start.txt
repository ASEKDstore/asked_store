/**
 * Handle /start command
 */
export async function handleStart(ctx: Context) {
  try {
    const user = ctx.from
    if (!user) {
      return ctx.reply('РћС€РёР±РєР°: РЅРµ СѓРґР°Р»РѕСЃСЊ РїРѕР»СѓС‡РёС‚СЊ РґР°РЅРЅС‹Рµ РїРѕР»СЊР·РѕРІР°С‚РµР»СЏ')
    }

    // Register/update user
    saveUser({
      id: user.id,
      username: user.username,
      first_name: user.first_name,
      last_name: user.last_name,
    })

    // Get video source
    const videoSource = getWelcomeVideoSource()
    const caption = getWelcomeCaption(user)
    const keyboard = getMainMenuKeyboard()

    // Send video with caption and keyboard
    if (videoSource) {
      // If it's a URL, use it directly
      if (videoSource.startsWith('http://') || videoSource.startsWith('https://')) {
        await ctx.replyWithVideo(videoSource, {
          caption,
          reply_markup: keyboard,
        })
      } else {
        // If it's a local file path, use source
        await ctx.replyWithVideo({ source: videoSource }, {
          caption,
          reply_markup: keyboard,
        })
      }
    } else {
      // Fallback: send text message if no video
      await ctx.reply(caption, {
        reply_markup: keyboard,
      })
    }

    // Demo: РѕС‚РїСЂР°РІРєР° СЃС‚Р°С‚СѓСЃРЅС‹С… СЃРѕРѕР±С‰РµРЅРёР№ СЃ РєР°СЃС‚РѕРјРЅС‹РјРё СЌРјРѕРґР·Рё
    try {
      await status.appStarting(ctx.telegram, ctx.chat.id)
      await status.labStarting(ctx.telegram, ctx.chat.id)
      await status.checkPayment(ctx.telegram, ctx.chat.id)
      await status.paymentOk(ctx.telegram, ctx.chat.id)
    } catch (error) {
      console.error('вќЊ Error sending status messages:', error)
      // РќРµ РїСЂРµСЂС‹РІР°РµРј РІС‹РїРѕР»РЅРµРЅРёРµ, РµСЃР»Рё РѕС€РёР±РєР° РІ РґРµРјРѕ-СЃРѕРѕР±С‰РµРЅРёСЏС…
    }
  } catch (error) {
    console.error('вќЊ Error in handleStart:', error)
    await ctx.reply('РџСЂРѕРёР·РѕС€Р»Р° РѕС€РёР±РєР° РїСЂРё РѕР±СЂР°Р±РѕС‚РєРµ РєРѕРјР°РЅРґС‹. РџРѕРїСЂРѕР±СѓР№С‚Рµ РїРѕР·Р¶Рµ.')
  }
}
